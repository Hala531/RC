CO2 %>% 
  ggplot(aes(conc, uptake, color = Treatment))+
  geom_point(size = 3, alpha = 0.7) + 
  geom_smooth(method = lm, se = F)+
  #facet_wrap() est une fonction dans ggplot2 en R qui permet de diviser un graphique en plusieurs sous-graphiques, appels facettes, en fonction d'une variable catgorielle.
  facet_wrap(~Type)+
  labs(title = "Concentration of co2") +
  theme_bw()

# theme_bw() est une fonction dans ggplot2 qui applique un thme pr-construit  votre graphique, modifiant l'apparence visuelle gnrale. Le thme bw signifie "black and white" (noir et blanc), et il est conu pour donner un style pur et professionnel  votre graphique.
